import fetch from 'node-fetch'; // npm i node-fetch if missing

const TOKEN = process.env.LOYVERSE_ACCESS_TOKEN;
const WEBHOOK_URL = 'https://your-replit-domain.replit.app/api/loyverse-webhook'; // Replace with your domain
const SECRET = process.env.LOYVERSE_WEBHOOK_SECRET;

async function createWebhook() {
  try {
    const response = await fetch('https://api.loyverse.com/v1.0/webhooks', {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${TOKEN}`,
        'Content-Type': 'application/json'
      },
      body: JSON.stringify({
        url: WEBHOOK_URL,
        secret: SECRET,
        events: ['receipts.created']
      })
    });
    const data = await response.json();
    console.log('Webhook created:', data);
  } catch (error) {
    console.error('Error creating webhook:', error);
  }
}

createWebhook();
Step 4: Loyverse Dashboard Setup (If Manual)
Log into Loyverse Back Office > Settings > Integrations > Webhooks > Add Webhook.
URL: Your Replit domain + /api/loyverse-webhook.
Secret: Copy from .env.
Events: Select "Receipts Created".
Save > Test (check Replit logs for receipt).
