import OpenAI from "openai";

const client = new OpenAI({
  apiKey: process.env.OPENAI_API_KEY,
});

export async function askGPT(prompt: string, system: string = "You are a helpful assistant.") {
  const response = await client.chat.completions.create({
    model: "gpt-4o",
    messages: [
      { role: "system", content: system },
      { role: "user", content: prompt },
    ],
  });
  return response.choices[0]?.message?.content ?? "";
}

10. server/routes/agents.ts
/**
 * ðŸš¨ DO NOT MODIFY THIS FILE WITHOUT CAM'S APPROVAL ðŸš¨
 * Routes for multi-agent chat + Jussi ops
 */

import express from "express";
import { jussiHandler } from "../agents/jussi";
import { ollieHandler } from "../agents/ollie";
import { sallyHandler } from "../agents/sally";
import { marloHandler } from "../agents/marlo";
import { bigbossHandler } from "../agents/bigboss";
import { generateJussiSummary } from "../services/jussiDailySummaryService";

const router = express.Router();

// Multi-agent chat
router.post("/chat/:agent", async (req, res) => {
  const { agent } = req.params;
  const { message } = req.body;

  try {
    let reply;
    switch (agent) {
      case "jussi": reply = await jussiHandler(message); break;
      case "ollie": reply = await ollieHandler(message); break;
      case "sally": reply = await sallyHandler(message); break;
      case "marlo": reply = await marloHandler(message); break;
      case "bigboss": reply = await bigbossHandler(message); break;
      default: reply = "Unknown agent.";
    }
    res.json({ reply });
  } catch (e) {
    res.status(500).json({ error: "Agent failed", details: e });
  }
});

// Jussi endpoints
router.get("/jussi/status", async (_req, res) => {
  res.json({ status: "ok", lastRun: new Date().toISOString() });
});

router.post("/jussi/generate", async (req, res) => {
  const { date, receipts } = req.body;
  const summary = await generateJussiSummary(date, receipts);
  res.json({ summary });
});

export default router;

/**
 * ðŸš¨ DO NOT MODIFY THIS FILE WITHOUT CAM'S APPROVAL ðŸš¨
 * JussiOps â€“ Dedicated Ops page
 */

import { useState } from "react";

export default function JussiOps() {
  const [input, setInput] = useState("");
  const [messages, setMessages] = useState<string[]>([]);

  async function sendMessage() {
    const res = await fetch("/chat/jussi", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ message: input }),
    });
    const data = await res.json();
    setMessages([...messages, "You: " + input, "Jussi: " + data.reply]);
    setInput("");
  }

  return (
    <div className="p-6">
      <h1 className="text-xl font-bold">Jussi Operations Chat</h1>
      <div className="mt-4 space-y-2">
        {messages.map((msg, i) => <div key={i} className="text-sm">{msg}</div>)}
      </div>
      <div className="mt-4 flex space-x-2">
        <input
          className="border p-2 flex-1"
          value={input}
          onChange={(e) => setInput(e.target.value)}
        />
        <button onClick={sendMessage} className="bg-yellow-400 px-4 py-2">Send</button>
      </div>
    </div>
  );
}
