    function goBack() {
      window.history.back();
    }

    async function loadStockItems() {
      const res = await fetch('/api/food-costings');
      const data = await res.json();

      const categories = {
        'Drink Stock': 'drink-stock',
        'Fresh Food': 'fresh-food',
        'Frozen Food': 'frozen-food',
        'Shelf': 'shelf-items',
        'Kitchen': 'kitchen-items',
        'Packaging': 'packaging-items',
      };

      Object.keys(categories).forEach(category => {
        const div = document.getElementById(categories[category]);
        const filtered = data.filter(i => i.Category === category);
        filtered.forEach(item => {
          const label = document.createElement('label');
          label.innerHTML = `${item.Item}: <input type="number" name="${item.Item}" min="0">`;
          div.appendChild(label);
        });
      });
    }

    loadStockItems();
  </script>
